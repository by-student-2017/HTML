Å† Install (gfortran + OpenBLAS + Scalapack + OpenMPI + python 2.7) 
(Attension!!!: This installation is test version) (Edit: 31/May/2020) 
0. sudo apt update
  (sudo apt --fix-broken install)
1. sudo apt -y install wget python3-dev gfortran libopenblas-dev mpi-default-bin mpi-default-dev ssh make python libscalapack-openmpi-dev
2. wget https://github.com/nwchemgit/nwchem/archive/v7.0.0-release.tar.gz
3. tar zxvf v7.0.0-release.tar.gz
4. cd nwchem-7.0.0-release
5. export NWCHEM_TOP=$HOME/nwchem-7.0.0-release
  export USE_MPI=y
  export NWCHEM_TARGET=LINUX64
  export USE_PYTHONCONFIG=y
  export PYTHONVERSION=2.7
  export PYTHONHOME=/usr
  export BLAS_SIZE=4
  export USE_64TO32=y
  export BLASOPT="-lopenblas -lpthread -lrt"
  export LAPACK_LIB="-L/usr/lib/x86_64-linux-gnu/openblas -llapack -lblas"
  export SCALAPACK_SIZE=4
  export USE_SCALAPACK=y
  export SCALAPACK="-lscalapack-openmpi"
  export LD_LIBRARY_PATH=$LD_LIBRARY_PATH:/usr/lib/x86_64-linux-gnu:/usr/lib/x86_64-linux-gnu/openmpi/lib
  export FC=mpif90
  export CC=mpicc
4. cd $NWCHEM_TOP/src
5. cp ./config/make_nwchem_config ./config/nwchem_config.h
6. make nwchem_config NWCHEM_MODULES=all
7. make 64_to_32
8. make >& make.log
Å¶ "BLAS_SIZE=4" is related with "USE_64TO32=y" and "make 64_to_32."
Å¶ cpu (avx): export FFLAGS="-mavx"
Å¶ cpu (sse4.2): export FFLAGS="-msse4.2"

Å† examples (QMD, 3carbo_dft.nw)
0. export LD_LIBRARY_PATH=$LD_LIBRARY_PATH:/usr/lib/x86_64-linux-gnu:/usr/lib/x86_64-linux-gnu/openmpi/lib
1. cd ~/nwchem-7.0.0-release/examples
2. cd qmd
3. export OMP_NUM_THREADS=1
4. mpirun -np 2 $NWCHEM_TOP/bin/LINUX64/nwchem 3carbo_dft.nw | tee 3carbo_dft.out
