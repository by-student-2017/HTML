5. ./configure -prefix $PWD/qtbase -opensource -nomake tests

□ Install (adpt_core) (failed)
1. sudo apt update
  sudo apt install -y gcc build-essential
  sudo apt install -y cmake
  sudo apt install -y qt5-default
  sudo apt install -y libcurl4-openssl-dev
  sudo apt install -y libxml2-dev
  sudo apt install -y libblas-dev liblapack-dev liblapacke-dev gfortran
2. git clone https://bitbucket.org/solccp/adpt_core.git
3. cd adpt_core
4. git submodule init
5. git submodule update
6. cd external_libs
7. ./make-external-libs.sh
8. mkdir adpt_core_build
9. cd adpt_core_build
10. qmake PREFIX=~/adpt_core ~/adpt_core
11. make
12. sudo make install

[q1] Ubuntu 18.04.1 LTS にQt5 をインストールする
  https://qiita.com/chlochan/items/53f1276522d03f460a81
1. wget http://download.qt.io/official_releases/qt/5.11/5.11.2/qt-opensource-linux-x64-5.11.2.run
2. chmod +x qt-opensource-linux-x64-5.11.2.run
3. ./qt-opensource-linux-x64-5.11.2.run
4. sudo apt install -y build-essential
5. sudo apt install -y libfontconfig1
6. sudo apt install -y mesa-common-dev
7. sudo apt install -y libglu1-mesa-dev
8. cd ~/.local/share/applications
8. gedit Qt-Creator.desktop
-----
[Desktop Entry]
Version=1.0
Encoding=UTF-8
Type=Application
Name=QtCreator
Comment=QtCreator
NoDsiplay=true
Exec=/home/username/Qt5.11.2/Tools/QtCreator/bin/qtcreator %f
Icon=/home/username/Qt5.11.2/5.11.2/Src/qtdoc/doc/images/landing/icon_QtCreator_78x78px.png
Name[en_US]=Qt-Creator
-----
9. gedit defaults.list
-----
text/qtcreator=Qt-Creator.desktop;
-----
10. sudo gedit /usr/share/applications/mimeapps.list
-----[added Associations]
application/vnd.nokia.qt.qmakeprofile=qtcreator.desktop
-----
