Å† Install
0. cd ~
1. wget https://versaweb.dl.sourceforge.net/project/qvasp/qvasp-v2.21.tar.gz
2. tar zxvf qvasp-v2.21.tar.gz
3. cd qvasp-v2.21
 
Å† environment settings (easy method)
1. cd ~/qvasp-v2.21
2. chmod +x install.sh
3. ./install.sh
  do you wish the installer to prepend the qvasp location to PATH in /home/username/.bashrc ? (yes or no):yes
4. bash
 
Å† environment settings (other method)
1. gedit ~/.bashrc
  export qvasppath=$HOME/qvasp-v2.21
  export PATH=$qvasppath:$PATH
  export PATH=$qvasppath/exefile/Tools/USERTooLs/vtstscripts:$PATH
2. bash
 
Å† set potentials (no check)
1. cd $qvasppath/exefile/POTCAR
2. cp ~/vasp/potpaw_PBE/* ./paw_pbe
3. cp ~/vasp/potpaw_LDA/* ./paw_lda

Å† help
1. qvasp -help

Å† HSE calculation (no check)
1. qvasp -c2p
  (Transfer cif file to POSCAR)
2. qvasp -pbe ELE_Name
  (POTCAR from PBE)
3. qvasp -cp
  (Check the POTCAR)
4. qvasp -k 0.076
  (Create KPOINTS(Auto mesh))
5. qvasp -relax
  (For Structure Optimization calculation)
  (qvasp -p2c) (you can get cif file)
6. export OMP_NUM_THREADS=1
  mpirun -np 4 ~/vasp.5.4.1/bin/vasp_std
7. qvasp -hse
  (For HSE06 calculation)
8. export OMP_NUM_THREADS=1
  mpirun -np 4 ~/vasp.5.4.1/bin/vasp_std
9. qvasp -e
  (Read energy from OUTCAR)

Å† structure optimization + SCF calculation (no check)
1. qvasp -c2p
  (Transfer cif file to POSCAR)
2. qvasp -pbe ELE_Name
  (POTCAR from PBE)
3. qvasp -cp
  (Check the POTCAR)
4. qvasp -k 0.0764
  (Create KPOINTS(Auto mesh))
5. qvasp -scf 
  (For Self_Consistent calculation)
  (For Structure Optimization calculation)

Å† dos (no check)
6. qvasp -k 0.0764
  (Create KPOINTS(Auto mesh))
7. qvasp -dos
  (For Density Of States calculation)
8. export OMP_NUM_THREADS=1
  mpirun -np 4 ~/vasp.5.4.1/bin/vasp_std
9. qvasp -dosd
  (Deal DOS data from DOSCAR)
10. qvasp -ldos 1 2   
  (Get LDOS data from DOS1 DOS2.)

Å† band (no check)
6. gedit KPOINTS
  (write "Line-mode" information)
7. qvasp -bandd 
  (Deal Bands data from EIGENVAL)
8. export OMP_NUM_THREADS=1
  mpirun -np 4 ~/vasp.5.4.1/bin/vasp_std
9. qvasp -bandd
  (Deal Bands data from EIGENVAL)

Å† work function (no check)
