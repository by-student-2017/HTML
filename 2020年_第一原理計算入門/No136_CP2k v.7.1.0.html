Å† cp2k (latest version, this installation is not ver.7 but ver.8) (github)
(https://www.cp2k.org/exercises:2019_uzh_acpc2:installation)
(Attention!!!: This installation is test version) (2/Jun/2020)
0. sudo apt update
  sudo apt install -y wget make cmake gfortran g++ gcc build-essential libopenblas-dev libfftw3-dev libfftw3-mpi-dev libopenmpi-dev libscalapack-openmpi-dev python python3 pkg-config libxc-dev libgsl-dev gsl-bin libhdf5-dev libhdf5-openmpi-dev git bison byacc libint-dev libint2-dev libelpa-dev 
1. git clone --recursive https://github.com/cp2k/cp2k.git cp2k
2. cd cp2k/arch
3. gedit ubuntu-x86-64-gfortran.psmp
-----
CC          = mpicc
FC          = mpif90
LD          = mpif90
AR          = ar -r

CFLAGS      = -O2 -g -mtune=native

DFLAGS      = -D__FFTW3
DFLAGS     += -D__MPI_VERSION=3
DFLAGS     += -D__parallel -D__SCALAPACK

FCFLAGS     = $(CFLAGS) $(DFLAGS)
FCFLAGS    += -ffree-form -ffree-line-length-none
FCFLAGS    += -fopenmp
FCFLAGS    += -ftree-vectorize -funroll-loops -std=f2008

LDFLAGS     = $(FCFLAGS) -static-libgfortran

LIBS       += -lxcf03
LIBS       += -lfftw3
LIBS       += -lfftw3_threads
LIBS       += -lscalapack-openmpi
LIBS       += -lopenblas
LIBS       += -ldl -lpthread -lstdc++
-----
4. cd ~/cp2k 
5. make -j 2 ARCH=ubuntu-x86-64-gfortran VERSION=psmp
6. gedit ~/.bashrc
-----
export PATH=${PATH}:$HOME/cp2k/exe/ubuntu-x86-64-gfortran
-----
7. source ~/.bashrc
8. export OMP_NUM_THREADS=1
9. mpirun -np 2 ./cp2k.psmp --version
